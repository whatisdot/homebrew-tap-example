def parse_url_pattern
  unless match = url.match(%r{https://github.com/([^/]+)/([^/]+)(/\S*)*})
    raise CurlDownloadStrategyError, 'Invalid url pattern for GitHub Repository.'
  end

  _, @owner, @repo, @filepath = *match
end
